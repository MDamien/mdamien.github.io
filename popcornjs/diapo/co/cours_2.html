Démonstration

Popcorn.js est concrètement une librairie Javascript, nous pouvons donc facilement avoir un premier exemple fonctionnel

<!doctype html>
<html>
  <head>
    <script src="http://popcornjs.org/code/dist/popcorn-complete.min.js"></script>
  </head>
  <body>
    <video height="180" width="300" id="ourvideo" controls>
      <source src="http://videos.mozilla.org/serv/webmademovies/popcornplug.ogv">
    </video>
    <div id="footnotediv"></div>
    <script>
         var pop = Popcorn("#ourvideo");
         pop.footnote({
             start: 2,
             end: 6,
             text: "Pop!",
             target: "footnotediv"
         }); 
         pop.play();
    </script>
  </body>
</html>
